Create procedure [dbo].[sp_SaveExpenditure] 
		@Id int,
		@Name nvarchar(25),
		@Sum decimal(5,2)
AS
BEGIN	
	SET NOCOUNT ON;
	SET IDENTITY_INSERT Expenditures Off
	if(@Id > 0)
	begin
		Update Expenditures
		SET 
		Name =@Name,
		Sum = @Sum
		WHERE ExpenditureId =@Id
 	end
	else
    BEGIN 	
		  INSERT Expenditures
		  (
			  Name,
			  Sum
		  )
		  Values
		  (			 
			  @Name,
			  @Sum			 
		  )	
             Select SCOPE_IDENTITY() As ExpenditureId	 	 
	END	
	
END
