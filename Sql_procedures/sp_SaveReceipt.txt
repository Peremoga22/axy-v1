Create procedure [dbo].[sp_SaveReceipt] 
		@Id int,
		@Name nvarchar(25),
		@Sum decimal(5,2)
AS
BEGIN	
	SET NOCOUNT ON;
	SET IDENTITY_INSERT Receipts Off
	if(@Id > 0)
	begin
		Update Receipts
		SET 
		Name =@Name,
		Sum = @Sum
		WHERE ReceiptId =@Id
 	end
	else
    BEGIN 	
		  INSERT Receipts 
		  (
			  Name,
			  Sum
		  )
		  Values
		  (			 
			  @Name,
			  @Sum			 
		  )	
		Select SCOPE_IDENTITY() As ReceiptId	 
	END	
	
END
