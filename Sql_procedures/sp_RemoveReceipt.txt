Create procedure [dbo].[sp_RemoveReceipt] 
		@Id int		
AS
BEGIN	
	SET NOCOUNT ON;	
    BEGIN 	
		 Delete from Receipts
		 where ReceiptId = @Id 
	END	
	
END
